## app.component.ts> 

import { Component } from '@angular/core';
@Component({
  selector: 'app-root',
  templateUrl: './app.component.html'
})
export class AppComponent {
  name: string = '';
  address: string = '';
  contact: string = '';
  email: string = '';
  showConfirmation: boolean = false;

  register() {
    this.showConfirmation = true;
  }
}
